\subsection{General Laws}
    \begin{itemize}[label=-]
        \item Biot-Savart \& Lorentz
            $$ \boldsymbol{F} = I (\boldsymbol{l} \times \boldsymbol{B}) = q (\boldsymbol{v} \times \boldsymbol{B}) $$
        \item Faraday
            $$ U = - v (\boldsymbol{l} \times \boldsymbol{B})$$
        \item Motor
            $$ T(t) = \kappa \cdot I(t)$$
        \item Generator
            $$ U_{ind}(t) = \kappa \cdot \omega(t)$$
    \end{itemize}
